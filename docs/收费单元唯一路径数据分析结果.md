## 1. 数据字典

| 字段名 | 字段格式 | 字段描述 |
|--------|----------|----------|
| id | varchar(20) NOT NULL | 收费单元编号 |
| name | varchar(50) NULL | 收费单元名称 |
| section_number | int(11) NULL | 收费单元所在串行路段的编号，同一个编号的收费单元说明再同一个串行路段上 |
| type | int(11) NULL | 所在路段性质(2-还贷、1-经营) |
| length | int(11) NULL | 起止里程 |
| startLat | varchar(20) NULL | 起始计费位置纬度 |
| startLng | varchar(20) NULL | 起始计费位置经度 |
| startStakeNum | decimal(10,3) NULL | 起始计费位置桩号 |
| endStakeNum | decimal(10,3) NULL | 终止计费位置桩号 |
| endLat | varchar(20) NULL | 终止计费位置纬度 |
| endLng | varchar(20) NULL | 终止计费位置经度 |
| tollRoads | varchar(150) NULL | 重合收费公路编号 |
| endTime | date NULL | 停止收费时间 |
| provinceType | int(11) NULL | 省界标识(0-非省界，1省界) |
| operation | int(1) NULL | 操作(1-新增,2-变更,3-删除) |
| isLoopCity | int(11) NULL | 绕城标识(0-非绕城,1-西安绕城,2-宝鸡绕城) |
| enTollStation | varchar(14) NULL | 入口收费站14位编码 |
| exTollStation | varchar(14) NULL | 出口收费站14位编码 |
| entrystation | int(11) NULL | 入口收费站原编码 |
| exitstation | int(11) NULL | 出口收费站原编码 |
| tollGrantry | varchar(16) NULL | 对应收费门架编号（含收费站作为虚拟门架） |
| ownerid | int(11) NULL | 所属路段业主 |
| roadid | int(11) NULL | 所属路段小业主编号 |
| roadidname | varchar(20) NULL | 所属路段小业主名称 |
| roadtype | int(11) NULL | 路段类型（1-普通公路；2-桥隧加收；3-西铜高速；4-一二级路） |
| feeKtype | int(11) NULL | 客车费率（1:甲类；2:乙类；3:丙类；4:丁类；5:戊类；6:己类；7:庚类；8:辛类；9:壬类；10:癸类） |
| feeHtype | int(11) NULL | 货车费率（1:甲类；2:乙类；3:丙类；4:丁类；5:戊类；6:己类；7:庚类；8:辛类；9:壬类；10:癸类） |
| status | int(2) NULL | 状态（1-在用；0-停用） |
| Gantrys | varchar(2000) NULL | 单元所有收费门架 |
| inoutprovince | varchar(2) NULL | 1出本省,2入本省 |
| HEX | varchar(6) NULL | 从1开始编号对应的HEX码 |
| NOTE | varchar(6) NULL | 普通0；省界1 ；出省11 入省12；虚拟2；二级公路3；西铜开放4，费显单元6 |
| SORT | int(11) NULL | 编号，从1开始 |
| DIRECTION | int(1) NOT NULL | 上下行(1-上行 2-下行) |
| BEGINTIME | date NOT NULL | 开始收费时间 |
| VERTICALSECTIONTYPE | int(1) NOT NULL | 收费单元类型(1-实体，2-虚拟) |
| tollstaion | varchar(20) NULL | 省界对应的收费站，后台维护 |
